
(define (domain grapevine)
    (:agents a b c d) 
    (:requirements :typing)
    (:types loc)
    (:anceff kd45closure
        :cond1 
        ; (
        ;     :poscond ?pos1
        ;     :negcond ?neg1
        ;     :rml [?a](?rml)
        ;     :type add
        ; )
        ; :cond2 
        ; (
        ;     :poscond ?pos2
        ;     :negcond ?neg2
        ;     :rml <?a>(?rml)
        ;     :type add
        ; )
    )

    (:predicates
        (at ?agent - agent ?l - loc)
        (connected ?l1 ?l2 - loc)
        {AK}(at2 ?agent - agent ?l - loc)
        {AK}(connected2 ?l1 ?l2 - loc)
    )

    (:action move
        ; :derive-condition   always
        :parameters         (?l1 ?l2 - loc) 
        :precondition       (and [?a](at ?a ?l1) (connected ?l1 ?l2))
        :effect             (and (at ?a ?l2))
    )
)
